import numpy as np

class QAgent:
    def __init__(self, states, actions):
        self.q = np.zeros((states, actions))

    def choose(self, s):
        return np.argmax(self.q[s])

    def update(self, s, a, r, ns):
        self.q[s][a] += 0.1 * (r + np.max(self.q[ns]) - self.q[s][a])
